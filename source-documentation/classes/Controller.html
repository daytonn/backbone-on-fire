    Optional methods and properties belonging to the Controller constrcutor's prototype.

  route: ['show/:id', 'edit/:id'],

  index: function() {
    // do index stuff
  },

  show: function(id) {
    // do show stuff
  },

  edit: function(id) {
    // do edit stuff
  },
});
